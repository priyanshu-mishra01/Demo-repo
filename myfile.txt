@RestController  
@RequestMapping("/api/students")  
public class StudentController {  
   
  @Autowired  
  private StudentService studentService;  
   
  @GetMapping  
  public List<Student> getAllStudents() {  
   return studentService.getAllStudents();  
  }  
   
  @GetMapping("/{id}")  
  public Student getStudentById(@PathVariable Long id) {  
   return studentService.getStudentById(id);  
  }  
   
  @GetMapping("/by-name/{firstName}/{lastName}")  
  public Student getStudentByFullName(@PathVariable String firstName, @PathVariable String lastName) {  
   return studentService.getStudentByFullName(firstName, lastName);  
  }  
   
  @GetMapping("/elective-subject/{studentId}")  
  public Subject getElectiveSubject(@PathVariable Long studentId) {  
   return studentService.getElectiveSubject(studentId);  
  }  
   
  @PostMapping("/assign-mentor")  
  public void assignMentor(@RequestBody AssignMentorRequest request) {  
   studentService.assignMentor(request.getStudentId(), request.getMentorId());  
  }  
   
  @PostMapping("/add-subject")  
  public void addSubject(@RequestBody AddSubjectRequest request) {  
   studentService.addSubject(request.getStudentId(), request.getSubjectId());  
  }  
   
  @PostMapping  
  public void createStudent(@RequestBody Student student) {  
   studentService.validateStudent(student);  
   studentService.save(student);  
  }  
}
